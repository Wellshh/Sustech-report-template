\section{Flowchart}

\begin{figure}[htbp]
  \centering
  \begin{sustechdiagram}
    \node[sustech terminator] (start) {Start};
    \node[sustech io, below=of start]           (input)  {Read Input};
    \node[sustech process, below=of input]      (proc)   {Preprocess};
    \node[sustech decision, below=14mm of proc] (check)  {Valid?};
    \node[sustech process, below left=14mm and 20mm of check]  (retry) {Retry};
    \node[sustech process, below right=14mm and 20mm of check] (run)   {Execute};
    \node[sustech datastore, below=of run]      (store)  {Save Result};
    \node[sustech terminator, below=of store]   (finish) {End};

    \sustechconnect{start}{input}
    \sustechconnect{input}{proc}
    \sustechconnectwith{sustech highlight arrow}{proc}{check}
    \sustechconnecttowith{sustech feedback arrow}[bend right=18][No]{check}{retry}
    \sustechconnecttowith{sustech highlight arrow}[bend left=18][Yes]{check}{run}
    \sustechconnect{run}{store}
    \sustechconnect{store}{finish}
    \sustechconnecttowith{sustech feedback arrow}[out=180,in=0,looseness=1.0][Loop]{retry}{input}
  \end{sustechdiagram}
  \caption{Flowchart benchmark figure}
\end{figure}
