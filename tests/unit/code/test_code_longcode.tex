"""
UniqueLongCode801 - this marker must appear in the rendered PDF.
Long code fixture to exercise the sustechlongcode environment.
"""
import sys

def step_one(data):
    return [x * 2 for x in data]

def step_two(data):
    return [x + 1 for x in data]

def step_three(data):
    total = sum(data)
    return total / max(len(data), 1)

def pipeline(raw):
    a = step_one(raw)
    b = step_two(a)
    return step_three(b)

if __name__ == "__main__":
    sample = list(range(20))
    result = pipeline(sample)
    print(f"Result: {result}")
    sys.exit(0)
